" removing backspace
" any character except CTRL+H followed by CTRL+H replace with nothing
":g/\([^^H]\)\(^H\)/s///g


" BackSpaceDelete
function! BackSpaceDelete()
    let lnum = 1
    while lnum < line('$')
        let l = getline(lnum)
        while l =~ '[^^H]^H'
            let l = substitute(l, '[^^H]^H', '', '')
        endwhile
        call setline(lnum, l)
        let lnum = lnum + 1
    endwhile
endfunction

command! -nargs=0 BSdelete call BackSpaceDelete()
