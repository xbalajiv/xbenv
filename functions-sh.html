# find pattern
fpn ()  
{
grep -l $1 $(find . -name "*.[ch]" -print)
}

# get user
guser ()
{
grep $1 /usr/local/etc/phonebook.lst | tr '~' '  '|cut -c1-70
}

# kill all processes except the current one
killme ()
{
kill -9 $(ps | grep -v $$ | grep -v grep | cut -c1-6 | tr '\n' ' ')
}

# kill the pine process
killpine ()
{
kill -9 $(psme | grep pine | grep -v grep | cut -c10-15)
}

# grep in history
hg ()
{
history | egrep $1
}

# find the files which has the said pattern($1), uses gid
files()
{
(gid32 $1 | cut -d ":" -f1-1 | uniq)
}

# search and edit the files which has the said pattern($1), uses gid
vid () 
{
    vim +/$1 $(files $1)
}

mkid()
{
    pushd .; cd source;
    find `pwd` -type f  -name "*.[chC]" -o -name "*.cpp"  > /tmp/flist.txt
#    find /home/xbalaji/main/source  -type f > /tmp/flist.txt
    mkid32 -dC++ -f/tmp/flist.txt -o ID
    \rm -f /tmp/flist.txt
    popd
}

lid () 
{
lid32 $1 | tr ' ' '\n'
}
